// --- Database Setup ---
async function initializeDatabase() {
  const dbPath = path.join(app.getPath("userData"), "fleet_modern_v2.db");

  db = await open({
    filename: dbPath,
    driver: sqlite3.Database,
  });

  // Ensure Tables Exist
  await db.exec(
    `CREATE TABLE IF NOT EXISTS vehicles (vehicle_id INTEGER PRIMARY KEY AUTOINCREMENT, make TEXT, model TEXT, year INTEGER, vin TEXT UNIQUE, license_plate TEXT, status TEXT, current_mileage INTEGER, department TEXT);`
  );
  await db.exec(
    `CREATE TABLE IF NOT EXISTS maintenance_logs (log_id INTEGER PRIMARY KEY AUTOINCREMENT, vehicle_id INTEGER, service_name TEXT, service_date TEXT, service_mileage INTEGER, description TEXT, cost REAL, performed_by TEXT, FOREIGN KEY (vehicle_id) REFERENCES vehicles (vehicle_id) ON DELETE CASCADE);`
  );
  await db.exec(
    `CREATE TABLE IF NOT EXISTS service_rules (rule_id INTEGER PRIMARY KEY AUTOINCREMENT, service_name TEXT UNIQUE, interval_miles INTEGER);`
  );

  // Initialize Rules (From Checklist)
  const rCount = await db.get("SELECT COUNT(*) as count FROM service_rules");
  if (rCount.count === 0) {
    const checklist = [
      { name: "Engine Oil", km: 10000 },
      { name: "Oil Filter", km: 10000 },
      { name: "Cabin AC Pollen Filter", km: 10000 },
      { name: "Alignment & Balancing", km: 10000 },
      { name: "Air Filter", km: 20000 },
      { name: "Fuel Filter", km: 20000 },
      { name: "Brake Oil", km: 20000 },
      { name: "Coolant", km: 20000 },
      { name: "Gear Transmission Oil", km: 40000 },
      { name: "Battery Water Check", km: 5000 },
      { name: "Tire Tread Inspection", km: 5000 },
    ];
    for (const item of checklist) {
      await db.run(
        "INSERT INTO service_rules (service_name, interval_miles) VALUES (?, ?)",
        item.name,
        item.km
      );
    }
  }
}

function getLocalIpAddress() {
  const interfaces = os.networkInterfaces();
  for (const name of Object.keys(interfaces)) {
    for (const iface of interfaces[name]) {
      if (iface.family === "IPv4" && !iface.internal) return iface.address;
    }
  }
  return "127.0.0.1";
}

// --- Server Logic ---
function startServer() {
  const expressApp = express();
  expressApp.use(cors());
  expressApp.use(express.json());

  // Alerts API
  expressApp.get("/api/alerts", async (req, res) => {
    try {
      const alerts = [];
      const vehicles = await db.all("SELECT * FROM vehicles");
      const rules = await db.all("SELECT * FROM service_rules");

      for (const v of vehicles) {
        for (const rule of rules) {
          const lastLog = await db.get(
            `SELECT * FROM maintenance_logs WHERE vehicle_id = ? AND service_name = ? ORDER BY service_mileage DESC LIMIT 1`,
            v.vehicle_id,
            rule.service_name
          );
          let kmSince = v.current_mileage;
          if (lastLog) kmSince = v.current_mileage - lastLog.service_mileage;

          if (kmSince >= rule.interval_miles) {
            alerts.push({
              vehicle_id: v.vehicle_id,
              vehicle_name: `${v.make} ${v.model}`,
              plate: v.license_plate,
              service_needed: rule.service_name,
              km_over: kmSince - rule.interval_miles,
              severity: "red",
            });
          } else if (kmSince >= rule.interval_miles - 1000) {
            alerts.push({
              vehicle_id: v.vehicle_id,
              vehicle_name: `${v.make} ${v.model}`,
              plate: v.license_plate,
              service_needed: rule.service_name,
              km_over: kmSince - rule.interval_miles,
              severity: "yellow",
            });
          }
        }
      }
      res.json(alerts);
    } catch (e) {
      res.status(500).json({ error: e.message });
    }
  });

  // Standard CRUD APIs
  expressApp.get("/api/vehicles", async (req, res) => {
    const vehicles = await db.all(
      "SELECT * FROM vehicles ORDER BY make, model"
    );
    res.json(vehicles);
  });
  expressApp.get("/api/vehicles/:id", async (req, res) => {
    const vehicle = await db.get(
      "SELECT * FROM vehicles WHERE vehicle_id = ?",
      req.params.id
    );
    if (vehicle) res.json(vehicle);
    else res.status(404).json({ error: "Not Found" });
  });
  expressApp.get("/api/vehicles/:id/logs", async (req, res) => {
    const logs = await db.all(
      "SELECT * FROM maintenance_logs WHERE vehicle_id = ? ORDER BY service_date DESC",
      req.params.id
    );
    res.json(logs);
  });
  expressApp.get("/api/service-rules", async (req, res) => {
    const rules = await db.all("SELECT * FROM service_rules");
    res.json(rules);
  });

  expressApp.post("/api/vehicles", async (req, res) => {
    const {
      make,
      model,
      year,
      vin,
      license_plate,
      status,
      current_mileage,
      department,
    } = req.body;
    try {
      const result = await db.run(
        "INSERT INTO vehicles (make, model, year, vin, license_plate, status, current_mileage, department) VALUES (?, ?, ?, ?, ?, ?, ?, ?)",
        make,
        model,
        year,
        vin,
        license_plate,
        status,
        current_mileage,
        department || "General"
      );
      res.status(201).json({ newId: result.lastID });
    } catch (e) {
      res.status(500).json({ error: e.message });
    }
  });

  expressApp.post("/api/vehicles/:id/logs", async (req, res) => {
    const {
      service_name,
      service_date,
      service_mileage,
      description,
      cost,
      performed_by,
    } = req.body;
    const vehicle_id = req.params.id;
    try {
      const result = await db.run(
        "INSERT INTO maintenance_logs (vehicle_id, service_name, service_date, service_mileage, description, cost, performed_by) VALUES (?, ?, ?, ?, ?, ?, ?)",
        vehicle_id,
        service_name,
        service_date,
        service_mileage,
        description,
        cost,
        performed_by
      );
      await db.run(
        "UPDATE vehicles SET current_mileage = ? WHERE vehicle_id = ? AND current_mileage < ?",
        service_mileage,
        vehicle_id,
        service_mileage
      );
      res.status(201).json({ newId: result.lastID });
    } catch (e) {
      res.status(500).json({ error: e.message });
    }
  });

  server = expressApp.listen(SERVER_PORT, "0.0.0.0", () =>
    console.log(`Server ON: ${SERVER_PORT}`)
  );
}

function createMainWindow(mode, serverIp = "") {
  mainWindow = new BrowserWindow({
    width: 1400,
    height: 900,
    minWidth: 1024,
    minHeight: 700,
    webPreferences: { preload: path.join(__dirname, "preload.js") },
    autoHideMenuBar: true,
  });

  const config = {
    mode: mode,
    apiUrl:
      mode === "server"
        ? `http://127.0.0.1:${SERVER_PORT}`
        : `http://${serverIp}:${SERVER_PORT}`,
    serverIp: getLocalIpAddress(),
  };

  mainWindow.loadURL(
    `data:text/html;charset=utf-8,${encodeURIComponent(getHtmlApp())}`
  );
  mainWindow.webContents.on("did-finish-load", () =>
    mainWindow.webContents.send("APP_CONFIG", config)
  );
}

app.on("ready", async () => {
  const { response } = await dialog.showMessageBox({
    type: "question",
    buttons: ["Main Server", "Client Node", "Cancel"],
    title: "App Mode",
    message: "Initialize System As:",
  });
  if (response === 0) {
    await initializeDatabase();
    startServer();
    createMainWindow("server");
  } else if (response === 1) {
    createMainWindow("client", "127.0.0.1");
  } else {
    app.quit();
  }
});

app.on("window-all-closed", () => {
  if (server) server.close();
  app.quit();
});

           el.classList.remove('hidden');
        }
        loadDashboard(); loadServiceRules();
      });
      async function api(endpoint, opts) { try { return await (await fetch(API_URL + endpoint, opts)).json(); } catch (e) { console.error(e); return null; } }
      
      async function loadServiceRules(filterType = 'all') {
        const rules = await api('/api/service-rules');
        const select = document.getElementById('l-name'); select.innerHTML = '';
        
        let filteredRules = rules;
        // Filter Logic: "Inspection" tasks have words like "Check", "Inspection". 
        // "Maintenance" tasks are everything else (Oil, Filters, etc).
        if (filterType === 'inspection') {
            filteredRules = rules.filter(r => r.service_name.includes('Check') || r.service_name.includes('Inspection') || r.service_name.includes('Alignment'));
        } else if (filterType === 'maintenance') {
            filteredRules = rules.filter(r => !r.service_name.includes('Check') && !r.service_name.includes('Inspection') && !r.service_name.includes('Alignment'));
        }

        filteredRules.forEach(r => select.innerHTML += \`<option value="\${r.service_name}">\${r.service_name} (Every \${r.interval_miles.toLocaleString()} Km)</option>\`);
        
        if (filterType === 'inspection' || filterType === 'all') select.innerHTML += \`<option value="General Inspection">General Inspection</option>\`;
        if (filterType === 'maintenance' || filterType === 'all') select.innerHTML += \`<option value="Unscheduled Repair">Unscheduled Repair</option>\`;
      }

      async function loadDashboard() {
        const alerts = await api('/api/alerts');
        const vehicles = await api('/api/vehicles');
        document.getElementById('stat-total').innerText = vehicles ? vehicles.length : 0;
        document.getElementById('stat-active').innerText = vehicles ? vehicles.filter(v => v.status === 'Active').length : 0;
        document.getElementById('stat-warning').innerText = alerts ? alerts.length : 0;
        document.getElementById('radar-blip').style.display = (alerts && alerts.length > 0) ? 'block' : 'none';

        const container = document.getElementById('alerts-container');
        container.innerHTML = '';
        if (!alerts || alerts.length === 0) container.innerHTML = '<div class="col-span-3 text-sm text-green-600 font-medium p-4 bg-green-50 border border-green-100 rounded-lg flex items-center justify-center gap-2">✅ All systems nominal.</div>';
        else {
           const grouped = {};
           alerts.forEach(a => {
             if(!grouped[a.vehicle_id]) grouped[a.vehicle_id] = { name: a.vehicle_name, issues: [], severity: 'yellow' };
             grouped[a.vehicle_id].issues.push(a.service_needed);
             if(a.severity === 'red') grouped[a.vehicle_id].severity = 'red';
           });

           for (const [id, g] of Object.entries(grouped)) {
             const border = g.severity === 'red' ? 'border-red-500' : 'border-yellow-500';
             const bg = g.severity === 'red' ? 'bg-red-50' : 'bg-yellow-50';
             container.innerHTML += \`
               <div onclick="loadDetails(\${id})" class="cursor-pointer border-l-4 \${border} \${bg} p-4 rounded-lg shadow-sm hover:shadow-md transition-shadow">
                 <div class="font-bold text-gray-800 text-lg">\${g.name}</div>
                 <div class="text-sm font-bold mt-1">\${g.issues.length} Items Due</div>
                 <div class="text-xs text-gray-600 mt-2">\${g.issues.join(', ')}</div>
               </div>\`;
           }
        }
      }

        const tbody = document.getElementById('fleet-table-body');

        if(!vehicles) return;

              <td class="px-6 py-4 text-gray-600">\${v.department || '-'}</td>

              <td class="px-6 py-4 font-mono text-gray-600">\${v.current_mileage.toLocaleString()}</td>
              <td class="px-6 py-4 text-right">
                <div onclick="event.stopPropagation()">
                   <select onchange="handleQuickAction(this, \${v.vehicle_id})" class="action-select text-xs border border-gray-300 rounded-md py-1 pl-2 pr-8 bg-white hover:border-sage-500 cursor-pointer outline-none focus:ring-2 focus:ring-sage-200 text-gray-600 font-medium">
                      <option value="">Actions</option>
                      <option value="log_maintenance">Log Maintenance</option>
                      <option value="log_inspection">Log Inspection</option>
                   </select>
                </div>
              </td>
            </tr>\`;
        });
      }
      function handleQuickAction(selectElem, id) {
         currentVehicleId = id;
         if(selectElem.value === 'log_maintenance') {
            openLogModal('maintenance');
         } else if(selectElem.value === 'log_inspection') {
            openLogModal('inspection');
         }
         selectElem.value = "";
      }
      async function loadDetails(id) {
        currentVehicleId = id;
        const [v, logs] = await Promise.all([api(\`/api/vehicles/\${id}\`), api(\`/api/vehicles/\${id}/logs\`)]);
        document.getElementById('detail-title').innerText = \`\${v.make} \${v.model}\`;
        document.getElementById('detail-plate').innerText = v.license_plate;
        document.getElementById('detail-vin').innerText = \`VIN: \${v.vin}\`;
        document.getElementById('detail-mileage').innerText = \`\${v.current_mileage.toLocaleString()} Km\`;
        const list = document.getElementById('log-list'); list.innerHTML = '';
        if(logs.length === 0) list.innerHTML = '<div class="p-8 text-center text-gray-400 italic">No history available.</div>';
        logs.forEach(l => {
           list.innerHTML += \`<div class="p-4 flex justify-between items-start hover:bg-gray-50 transition-colors"><div><div class="font-bold text-gray-800 text-sm">\${l.service_name}</div><div class="text-xs text-gray-500 mt-1">\${l.description || 'Routine Maintenance'}</div></div><div class="text-right"><div class="text-xs font-medium text-gray-500">\${l.service_date}</div><div class="text-xs text-gray-400 mt-1">\${l.service_mileage.toLocaleString()} Km • ₹\${l.cost || 0}</div></div></div>\`;
        });
        showPage('details');
      }
      function showPage(id) { document.querySelectorAll('.page').forEach(p => p.classList.remove('active')); document.getElementById(id + '-page').classList.add('active'); if(id==='home') loadDashboard(); if(id==='fleet') loadFleet(); }
      function openModal(id) { document.getElementById(id).classList.remove('hidden'); document.getElementById(id).classList.add('flex'); }
      function closeModal(id) { document.getElementById(id).classList.add('hidden'); document.getElementById(id).classList.remove('flex'); }
      
      function openLogModal(filterType = 'all') { 
         document.getElementById('l-date').valueAsDate = new Date(); 
         loadServiceRules(filterType); // Reload dropdown with specific filter
         openModal('log-modal'); 
      }

      document.getElementById('vehicle-form').onsubmit = async (e) => { e.preventDefault(); const data = { make: document.getElementById('v-make').value, model: document.getElementById('v-model').value, year: document.getElementById('v-year').value, vin: document.getElementById('v-vin').value, license_plate: document.getElementById('v-license').value, department: document.getElementById('v-dept').value, status: 'Active', current_mileage: document.getElementById('v-mileage').value }; await api('/api/vehicles', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify(data)}); closeModal('vehicle-modal'); showPage('fleet'); };
      document.getElementById('log-form').onsubmit = async (e) => { e.preventDefault(); const data = { service_name: document.getElementById('l-name').value, service_date: document.getElementById('l-date').value, service_mileage: parseInt(document.getElementById('l-mileage').value), description: document.getElementById('l-desc').value, cost: document.getElementById('l-cost').value, performed_by: 'Unit Mechanic' }; await api(\`/api/vehicles/\${currentVehicleId}/logs\`, { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify(data)}); closeModal('log-modal'); loadDetails(currentVehicleId); };
    </script>
